INTRODUCTION TO I2C COMMUNICATION

ADVANTAGES

  • Only uses two wires
  • Supports multiple masters and multiple slaves
  • ACK/NACK bit gives confirmation that each frame is transferred successfully
  • Well known and widely used protocol
  • Hardware is less complicated than with UARTs

DISADVANTAGES

  • Slower data transfer rate than SPI
  • The size of the data frame is limited to 8 bits
  • More complicated hardware needed to implement than SPI

SINGLE MASTER WITH MULTIPLE SLAVES

The problem with multiple masters in the same system comes when two masters try to send or receive data at the same time over the SDA line. To solve this problem, each master needs to detect if the SDA line is low or high before transmitting a message.

課堂練習 1 - Arduino UNO I2C Master Read Slave Write

https://www.arduino.cc/en/reference/wire

I2C Master

#include <Wire.h>

void setup() {
  Wire.begin(); 
  Serial.begin(9600);  
}

void loop() {
  Wire.requestFrom(4, 5);    
  
  Serial.print("Master read : ");
  while(Wire.available()) { 
    char c = Wire.read(); 
    Serial.print(c);         
  }
  Serial.println();

  delay(1000);
}

I2C Slave

#include <Wire.h>

void setup() {
  Serial.begin(9600);
  Wire.begin(4);  
  Wire.onRequest(requestEvent); 
}

char buf[6] = {'h','e','l','l','o',0};

void loop() {
  delay(100);
}

void requestEvent() {
  Serial.print("Slave write : ");
  Serial.println(buf);
  Wire.write(buf);
}

課堂練習 2 - Arduino UNO I2C Master Write Slave Read

https://www.arduino.cc/en/reference/wire

I2C Master

#include <Wire.h>

void setup() {
  Wire.begin();
  Serial.begin(9600);
}

char buf[10];

void i2buf(int v) {
  for(int i=4; i>=0; i--) {
  	buf[i] = v%10+'0';
    v/=10;
  }
  buf[5]=0;
}

void loop() {
  i2buf(random(10000,32767));
  Serial.print("Send => ");
  Serial.println(buf);
  Wire.beginTransmission(4); 
  Wire.write(buf);    
  Wire.endTransmission();          
  delay(2000);
}

I2C Slave

#include <Wire.h>

char buf[10];

void setup() {
  Serial.begin(9600);           
  Wire.begin(4);  
  Wire.onReceive(receiveEvent); 
}

void loop() {
  delay(100);
}

void receiveEvent(int howMany) {
  Serial.print(howMany);
  Serial.println(" bytes received");
  int i = 0;
  while(Wire.available()>0) {
    buf[i++] = Wire.read(); 
  }
  buf[i]=0;
  Serial.print("Received <= ");
  Serial.println(buf);
}

課堂練習 3 - Arduino UNO Scanning Available I2C Devices

https://www.arduino.cc/en/reference/wire

I2C Master

#include <Wire.h>
 
void setup(){
	Wire.begin();
	Serial.begin(9600);
	Serial.println("\nI2C Scanner");
}

void loop(){
  byte error, address;
  int nDevices;
  Serial.println("Scanning...");
  nDevices = 0;
  for (address = 1; address < 127; address++ ){
    Wire.beginTransmission(address);
    error = Wire.endTransmission();
    if (error == 0){
      Serial.print("I2C slave @ ");
      Serial.println(address);
      nDevices++;
    }else if (error == 4){
      Serial.print("Unknow error slave @ ");
      Serial.println(address);
    }
  }
  if (nDevices == 0)
  	Serial.println("No I2C slave found\n");
  else
  	Serial.println("done\n");
  delay(5000);
}

}

  delay(1000);
}

I2C Slave 2 (Dynamic Address) 

#include <Wire.h>

byte addr = (byte)random(1,10);

void setup() {
  Serial.begin(9600);
  randomSeed(analogRead(A0));
}

void loop() {
  Serial.print("addr = ");
  Serial.println(addr);
  Wire.begin(addr);
  delay(random(1,3)*1000);
  addr = (byte)random(1,10);
}

課程設計

Arduino Sketch rpi_i2c_lab01.ino

#include <Wire.h>

#define I2C_SLAVE_ADDRESS 11
#define PAYLOAD_SIZE 2

int cntdown = 0;

//----------------------------------------
void setup() {
  Wire.begin(I2C_SLAVE_ADDRESS);
  Serial.begin(9600); 
  Serial.println("--- I am Slave1 ---");
  delay(1000);               
  Wire.onRequest(requestEvents);
  Wire.onReceive(receiveEvents);
}

//----------------------------------------
void loop() {}

//----------------------------------------
void requestEvents() {
  Serial.print("sending value : ");
  while(cntdown>0) {
    int rn = random(100);
    Serial.print(rn);
    Serial.print(' ');
    Wire.write(rn);
    cntdown--;
  }
  Serial.println("\n");
}

//----------------------------------------
//----------------------------------------
void receiveEvents(int nb)
{  
    byte cmd;
    Serial.print("Num of bytes received : ");
    Serial.println(nb);
    if (Wire.available()) {
      cmd = Wire.read();
      if (cmd<=20) 
        cntdown = cmd;
      Serial.print("cmd : ");
      Serial.println(cmd);
    }
    if (nb>1) {
      Serial.print("Data received : ");
      while (Wire.available()) { 
        byte c = Wire.read();  
        switch(cmd) {
          case 123 : Serial.print(char(c));
                     break;
          case 111 : 
          default:
                     Serial.print(c);
                     Serial.print(" ");
                     break;
        }
      }
      Serial.println("\n");
    } 
}

import smbus2 as smbus 
from time import sleep
from random import randrange, random

I2C_ADDR = 11
PORT = 1

def i2c_write_bytes(data):
    cmd = 0
    if type(data) is type(""):
        data = data.encode("ascii")
        cmd = 123
    elif type(data) is type([]):
        cmd = 111
        data = bytes(data)
    i2cbus = smbus.SMBus(PORT)
    i2cbus.write_i2c_block_data(I2C_ADDR, cmd, data)
    sleep(0.5)

def toArduino(msg):
    i2c_write_bytes(msg)
    
if __name__ == '__main__':
    toArduino("STUST")
    toArduino([11,32,23,47,58])    
    toArduino("2021 Fall")

import smbus2 as smbus 
from time import sleep
from random import randrange, random

I2C_ADDR = 11
PORT = 1

def i2c_read_bytes(nb):
    i2cbus = smbus.SMBus(PORT)
    cmd = nb
    data = i2cbus.read_i2c_block_data(I2C_ADDR, cmd, nb)
    sleep(0.5)
    return data

def arduino2pi(n):
    recv = i2c_read_bytes(n)
    print(f"receved : {recv}")
    
if __name__ == '__main__':
    arduino2pi(6)
    arduino2pi(3)
    arduino2pi(8)

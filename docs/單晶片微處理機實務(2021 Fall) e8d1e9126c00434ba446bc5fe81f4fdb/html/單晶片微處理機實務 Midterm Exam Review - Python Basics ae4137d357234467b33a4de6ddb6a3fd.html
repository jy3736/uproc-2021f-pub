    cnt += 1

Snakify 範例解析 - Queens

def attact(a, b):
    c1 = a[0]==b[0] or a[1]==b[1]
    c2 = abs(a[0]-b[0])==abs(a[1]-b[1])
    return c1 or c2
    
q = []
for i in range(8):
    q.append([int(x) for x in input().split()])
    
chk = False
for i in range(7):
    for j in range(i+1,8):
        if attact(q[i],q[j]):
            chk = True
            break
		if chk: break
if chk:
    print("YES")
else:
    print("NO")
